4. Node.js 서버 구축해보기
```
const http = require('http');
const fs = require('fs');
const mySet = function(request, response) {
  let url = request.url;
  if (request.url === '/') {
    url = '/index.html';
  }
  if (request.url === '/about') {
    url = '/about.html';
  }
  response.writeHead(200);
  var htmlFile = fs.readFileSync(__dirname + url);
  response.end(htmlFile);
}
const app = http.createServer(mySet);
app.listen(8080);
```
